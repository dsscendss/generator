50      public String getFormattedContent() {
51          StringBuilder sb = new StringBuilder();
52  
53          for (String fileCommentLine : fileCommentLines) {
54              sb.append(fileCommentLine);
55              newLine(sb);
56          }
57  
58          if (getType().getPackageName() != null
59                  && getType().getPackageName().length() > 0) {
60              sb.append("package "); //$NON-NLS-1$
61              sb.append(getType().getPackageName());
62              sb.append(';');
63              newLine(sb);
64              newLine(sb);
65          }
66  
67          for (String staticImport : staticImports) {
68              sb.append("import static "); //$NON-NLS-1$
69              sb.append(staticImport);
70              sb.append(';');
71              newLine(sb);
72          }
73          
74          if (staticImports.size() > 0) {
75              newLine(sb);
76          }
77          
78          Set<String> importStrings = calculateImports(importedTypes);
79          for (String importString : importStrings) {
80              sb.append(importString);
81              newLine(sb);
82          }
83  
84          if (importStrings.size() > 0) {
85              newLine(sb);
86          }
87  
88          sb.append(super.getFormattedContent(0));
89  
90          return sb.toString();
91      }
