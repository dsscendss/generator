159     /**
160      * This is the main method for generating code. This method is long running,
161      * but progress can be provided and the method can be cancelled through the
162      * ProgressCallback interface.
163      * 
164      * @param callback
165      *            an instance of the ProgressCallback interface, or
166      *            <code>null</code> if you do not require progress information
167      * @param contextIds
168      *            a set of Strings containing context ids to run. Only the
169      *            contexts with an id specified in this list will be run. If the
170      *            list is null or empty, than all contexts are run.
171      * @param fullyQualifiedTableNames
172      *            a set of table names to generate. The elements of the set must
173      *            be Strings that exactly match what's specified in the
174      *            configuration. For example, if table name = "foo" and schema =
175      *            "bar", then the fully qualified table name is "foo.bar". If
176      *            the Set is null or empty, then all tables in the configuration
177      *            will be used for code generation.
178      * @throws InvalidConfigurationException
179      * @throws SQLException
180      * @throws IOException
181      * @throws InterruptedException
182      *             if the method is canceled through the ProgressCallback
183      */
184     public void generate(ProgressCallback callback, Set<String> contextIds,
185             Set<String> fullyQualifiedTableNames) throws SQLException,
186             IOException, InterruptedException {
187 
188         if (callback == null) {
189             callback = new NullProgressCallback();
190         }
191 
192         generatedJavaFiles.clear();
193         generatedXmlFiles.clear();
194 
195         // calculate the contexts to run
196         List<Context> contextsToRun;
197         if (contextIds == null || contextIds.size() == 0) {
198             contextsToRun = configuration.getContexts();
199         } else {
200             contextsToRun = new ArrayList<Context>();
201             for (Context context : configuration.getContexts()) {
202                 if (contextIds.contains(context.getId())) {
203                     contextsToRun.add(context);
204                 }
205             }
206         }
207 
208         // setup custom classloader if required
209         if (configuration.getClassPathEntries().size() > 0) {
210             ClassLoader classLoader = getCustomClassloader(configuration.getClassPathEntries());
211             ObjectFactory.addExternalClassLoader(classLoader);
212         }
213 
214         // now run the introspections...
215         int totalSteps = 0;
216         for (Context context : contextsToRun) {
217             totalSteps += context.getIntrospectionSteps();
218         }
219         callback.introspectionStarted(totalSteps);
220 
221         for (Context context : contextsToRun) {
222             context.introspectTables(callback, warnings,
223                     fullyQualifiedTableNames);
224         }
225 
226         // now run the generates
227         totalSteps = 0;
228         for (Context context : contextsToRun) {
229             totalSteps += context.getGenerationSteps();
230         }
231         callback.generationStarted(totalSteps);
232 
233         for (Context context : contextsToRun) {
234             context.generateFiles(callback, generatedJavaFiles,
235                     generatedXmlFiles, warnings);
236         }
237 
238         // now save the files
239         callback.saveStarted(generatedXmlFiles.size()
240                 + generatedJavaFiles.size());
241 
242         for (GeneratedXmlFile gxf : generatedXmlFiles) {
243             projects.add(gxf.getTargetProject());
244 
245             File targetFile;
246             String source;
247             try {
248                 File directory = shellCallback.getDirectory(gxf
249                         .getTargetProject(), gxf.getTargetPackage());
250                 targetFile = new File(directory, gxf.getFileName());
251                 if (targetFile.exists()) {
252                     if (gxf.isMergeable()) {
253                         source = XmlFileMergerJaxp.getMergedSource(gxf,
254                                 targetFile);
255                     } else if (shellCallback.isOverwriteEnabled()) {
256                         source = gxf.getFormattedContent();
257                         warnings.add(getString("Warning.11", //$NON-NLS-1$
258                                 targetFile.getAbsolutePath()));
259                     } else {
260                         source = gxf.getFormattedContent();
261                         targetFile = getUniqueFileName(directory, gxf
262                                 .getFileName());
263                         warnings.add(getString(
264                                 "Warning.2", targetFile.getAbsolutePath())); //$NON-NLS-1$
265                     }
266                 } else {
267                     source = gxf.getFormattedContent();
268                 }
269             } catch (ShellException e) {
270                 warnings.add(e.getMessage());
271                 continue;
272             }
273 
274             callback.checkCancel();
275             callback.startTask(getString(
276                     "Progress.15", targetFile.getName())); //$NON-NLS-1$
277             writeFile(targetFile, source, "UTF-8"); //$NON-NLS-1$
278         }
279 
280         for (GeneratedJavaFile gjf : generatedJavaFiles) {
281             projects.add(gjf.getTargetProject());
282 
283             File targetFile;
284             String source;
285             try {
286                 File directory = shellCallback.getDirectory(gjf
287                         .getTargetProject(), gjf.getTargetPackage());
288                 targetFile = new File(directory, gjf.getFileName());
289                 if (targetFile.exists()) {
290                     if (shellCallback.isMergeSupported()) {
291                         source = shellCallback.mergeJavaFile(gjf
292                                 .getFormattedContent(), targetFile
293                                 .getAbsolutePath(),
294                                 MergeConstants.OLD_ELEMENT_TAGS,
295                                 gjf.getFileEncoding());
296                     } else if (shellCallback.isOverwriteEnabled()) {
297                         source = gjf.getFormattedContent();
298                         warnings.add(getString("Warning.11", //$NON-NLS-1$
299                                 targetFile.getAbsolutePath()));
300                     } else {
301                         source = gjf.getFormattedContent();
302                         targetFile = getUniqueFileName(directory, gjf
303                                 .getFileName());
304                         warnings.add(getString(
305                                 "Warning.2", targetFile.getAbsolutePath())); //$NON-NLS-1$
306                     }
307                 } else {
308                     source = gjf.getFormattedContent();
309                 }
310 
311                 callback.checkCancel();
312                 callback.startTask(getString(
313                         "Progress.15", targetFile.getName())); //$NON-NLS-1$
314                 writeFile(targetFile, source, gjf.getFileEncoding());
315             } catch (ShellException e) {
316                 warnings.add(e.getMessage());
317             }
318         }
319 
320         for (String project : projects) {
321             shellCallback.refreshProject(project);
322         }
323 
324         callback.done();
325     }

327     /**
328      * Writes, or overwrites, the contents of the specified file
329      * 
330      * @param file
331      * @param content
332      */
333     private void writeFile(File file, String content, String fileEncoding) throws IOException {
334         FileOutputStream fos = new FileOutputStream(file, false);
335         OutputStreamWriter osw;
336         if (fileEncoding == null) {
337             osw = new OutputStreamWriter(fos);
338         } else {
339             osw = new OutputStreamWriter(fos, fileEncoding);
340         }
341         
342         BufferedWriter bw = new BufferedWriter(osw);
343         bw.write(content);
344         bw.close();
345     }

347     private File getUniqueFileName(File directory, String fileName) {
348         File answer = null;
349 
350         // try up to 1000 times to generate a unique file name
351         StringBuilder sb = new StringBuilder();
352         for (int i = 1; i < 1000; i++) {
353             sb.setLength(0);
354             sb.append(fileName);
355             sb.append('.');
356             sb.append(i);
357 
358             File testFile = new File(directory, sb.toString());
359             if (!testFile.exists()) {
360                 answer = testFile;
361                 break;
362             }
363         }
364 
365         if (answer == null) {
366             throw new RuntimeException(getString(
367                     "RuntimeError.3", directory.getAbsolutePath())); //$NON-NLS-1$
368         }
369 
370         return answer;
371     }
